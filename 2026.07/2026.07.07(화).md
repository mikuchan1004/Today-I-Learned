### 증감 연산자

- 값을 1씩 늘리거나 줄일 때 사용하는 연산자

- 사용할때는 덧셈이나 뺄셈 연산자를 두 개 연달아 붙여서 사용.

``` javascript
let a = 1;
a++; // 증감 연산자를 변수 뒤에 두면 증감 연산의 결과는 연산자가 있는 다음 행부터 반영. (후위 연산)
console.log(a); // output : 2

let b = 1;
b--;
console.log(b); // output : 8
```

``` javascript
let a = 1;
// 증감 연산자를 변수 앞에 두어 증감 연산을 작성한 행부터 적용하는 것을 전위 연산이라고 한다.
console.log(++a); // output : 2 
console.log(a); // output : 2

let b = 1;
console.log(--b); // output : 0
console.log(b); // output : 0
```
